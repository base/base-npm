## Usage

```js
var npm = require('{%= name %}');
var Base = require('base');
var app = new Base();
app.use(npm());

// install npm packages `micromatch` and `is-absolute`
app.npm.saveDev(['micromatch', 'is-absolute'], function(err) {
  if (err) throw err;
});
```
